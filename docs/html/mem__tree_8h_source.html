10/**
11 * @brief Represents a node in the generic heap tree, tracking a dynamically allocated memory block.
12 *
13 * Each node stores metadata about a memory allocation, including its pointer, size,
14 * expiration time, and reference count. It also includes flags for managing its
15 * lifecycle within the heap tree and a mutex for thread-safe access.
16 */
17typedef struct ttak_mem_node {
18 void *ptr; /**< Pointer to the actual memory block. */
19 size_t size; /**< Size of the allocated memory block. */
20 uint64_t expires_tick; /**< Monotonic tick when this memory block should expire. */
21 _Atomic uint32_t ref_count; /**< Atomic count of references to this node. */
22 _Bool is_root; /**< True if this node is referenced externally (not by another heap node). */
23 pthread_mutex_t lock; /**< Mutex for thread-safe access to this node's metadata. */
24 struct ttak_mem_node *next; /**< Pointer to the next node in the mem tree's internal list. */
25} ttak_mem_node_t;

27/**
28 * @brief Manages the collection of dynamically allocated memory blocks as a mem tree.
29 *
30 * This structure provides a centralized mechanism for tracking memory allocations,
31 * their interdependencies (via reference counts), and their lifetimes. It supports
32 * automatic cleanup of expired or unreferenced blocks and allows for manual control
33 * over the cleanup process.
34 */
35typedef struct ttak_mem_tree {
36 ttak_mem_node_t *head; /**< Head of the linked list of all tracked mem nodes. */
37 pthread_mutex_t lock; /**< Mutex for thread-safe access to the mem tree structure. */
38 _Atomic uint64_t cleanup_interval_ns; /**< Interval in nanoseconds for automatic cleanup. */
39 _Atomic _Bool use_manual_cleanup; /**< Flag to disable automatic cleanup (1 for manual, 0 for auto). */
40 pthread_t cleanup_thread; /**< Thread ID for the background automatic cleanup process. */
41 _Atomic _Bool shutdown_requested; /**< Flag to signal the cleanup thread to terminate. */
42} ttak_mem_tree_t;
