12/**
13 * @brief Represents a node in the generic heap tree, tracking a dynamically allocated memory block.
14 *
15 * Each node stores metadata about a memory allocation, including its pointer, size,
16 * expiration time, and reference count. It also includes flags for managing its
17 * lifecycle within the heap tree and a mutex for thread-safe access.
18 */
19typedef struct ttak_mem_node {
20 void *ptr; /**< Pointer to the actual memory block. */
21 size_t size; /**< Size of the allocated memory block. */
22 uint64_t expires_tick; /**< Monotonic tick when this memory block should expire. */
23 _Atomic uint32_t ref_count; /**< Atomic count of references to this node. */
24 _Bool is_root; /**< True if this node is referenced externally (not by another heap node). */
25 pthread_mutex_t lock; /**< Mutex for thread-safe access to this node's metadata. */
26 struct ttak_mem_node *next; /**< Pointer to the next node in the mem tree's internal list. */
27 struct ttak_mem_node *prev; /**< Pointer to the previous node in the mem tree's internal list. */
28 ttak_mem_tree_t *tree; /**< Pointer back to the parent mem tree. */
29} ttak_mem_node_t;

31/**
32 * @brief Manages the collection of dynamically allocated memory blocks as a mem tree.
33 *
34 * This structure provides a centralized mechanism for tracking memory allocations,
35 * their interdependencies (via reference counts), and their lifetimes. It supports
36 * automatic cleanup of expired or unreferenced blocks and allows for manual control
37 * over the cleanup process.
38 */
39struct ttak_mem_tree {
40 ttak_mem_node_t *head; /**< Head of the linked list of all tracked mem nodes. */
41 pthread_mutex_t lock; /**< Mutex for thread-safe access to the mem tree structure. */
42 pthread_cond_t cond; /**< Condition variable for immediate cleanup wakeup. */
43 _Atomic uint64_t max_cleanup_interval_ns; /**< Maximum interval in nanoseconds for automatic cleanup (default 120s). */
44 _Atomic uint64_t min_cleanup_interval_ns; /**< Minimum interval in nanoseconds for automatic cleanup (default 10s). */
45 _Atomic size_t garbage_pressure; /**< Score representing the amount of potential garbage/missing memory. */
46 _Atomic size_t pressure_threshold; /**< Threshold to trigger immediate cleanup (default 1MB). */
47 _Atomic _Bool use_manual_cleanup; /**< Flag to disable automatic cleanup (1 for manual, 0 for auto). */
48 pthread_t cleanup_thread; /**< Thread ID for the background automatic cleanup process. */
49 _Atomic _Bool shutdown_requested; /**< Flag to signal the cleanup thread to terminate. */
50};
