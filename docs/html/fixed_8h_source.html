129static inline void ttak_mul_64(uint64_t a, uint64_t b, uint64_t *hi, uint64_t *lo) {
130 /* Inline ISA primitives keep TinyCC builds fast without optimizer help. */
131#if defined(__x86_64__)
132 uint64_t lo_tmp, hi_tmp;
133 __asm__ __volatile__("mulq %[rhs]"
134 : "=a"(lo_tmp), "=d"(hi_tmp)
135 : "0"(a), [rhs] "r"(b)
136 : "cc");
137 *lo = lo_tmp;
138 *hi = hi_tmp;
139 return;
140#elif defined(__aarch64__)
141 uint64_t lo_tmp, hi_tmp;
142 __asm__ __volatile__(
143 "mul %0, %2, %3\n"
144 "umulh %1, %2, %3"
145 : "=&r"(lo_tmp), "=&r"(hi_tmp)
146 : "r"(a), "r"(b));
147 *lo = lo_tmp;
148 *hi = hi_tmp;
149 return;
150#elif defined(__riscv_xlen) && (__riscv_xlen == 64)
151 uint64_t lo_tmp, hi_tmp;
152 __asm__ __volatile__(
153 "mul %0, %2, %3\n"
154 "mulhu %1, %2, %3"
155 : "=&r"(lo_tmp), "=&r"(hi_tmp)
156 : "r"(a), "r"(b));
157 *lo = lo_tmp;
158 *hi = hi_tmp;
159 return;
160#endif
161
162 uint64_t a_lo = (uint32_t)a;
163 uint64_t a_hi = a >> 32;
164 uint64_t b_lo = (uint32_t)b;
165 uint64_t b_hi = b >> 32;
166
167 uint64_t p0 = a_lo * b_lo;
168 uint64_t p1 = a_lo * b_hi;
169 uint64_t p2 = a_hi * b_lo;
170 uint64_t p3 = a_hi * b_hi;
171
172 uint64_t mid = (p0 >> 32) + (p1 & 0xFFFFFFFFULL) + (p2 & 0xFFFFFFFFULL);
173 uint64_t mid_hi = (mid >> 32) + (p1 >> 32) + (p2 >> 32);
174
175 *lo = (mid << 32) | (p0 & 0xFFFFFFFFULL);
176 *hi = p3 + mid_hi;
177}

243 ttak_u256_t out;
244 uint64_t carry = 0;
245 for (int i = 0; i < 4; ++i) {
246 uint64_t sum = a.limb[i] + b.limb[i];
247 uint64_t new_sum = sum + carry;
248 uint64_t new_carry = (sum < a.limb[i]) || (new_sum < sum);
249 out.limb[i] = new_sum;
250 carry = new_carry;
251 }

255static inline ttak_u256_t ttak_u256_shr(ttak_u256_t v, unsigned shift) {
256 if (shift >= 256) return ttak_u256_zero();
257 if (shift == 0) return v;
258 ttak_u256_t out = ttak_u256_zero();
259 unsigned limb_shift = shift / 64;
260 unsigned bit_shift = shift % 64;
261 for (int i = 3; i >= 0; --i) {
262 if ((unsigned)i < limb_shift) continue;
263 uint64_t chunk = v.limb[i];
264 unsigned target = i - limb_shift;
265 out.limb[target] |= chunk >> bit_shift;
266 if (bit_shift && target > 0) {
267 out.limb[target - 1] |= chunk << (64 - bit_shift);
268 }
269 }
270 return out;

291static inline ttak_u256_t ttak_u128_mul_u128(ttak_u128_t a, ttak_u128_t b) {
292 uint64_t lo_hi, lo_lo;
293 ttak_mul_64(a.lo, b.lo, &lo_hi, &lo_lo);
294
295 uint64_t cross1_hi, cross1_lo;
296 ttak_mul_64(a.lo, b.hi, &cross1_hi, &cross1_lo);
297
298 uint64_t cross2_hi, cross2_lo;
299 ttak_mul_64(a.hi, b.lo, &cross2_hi, &cross2_lo);
300
301 uint64_t hi_hi, hi_lo;
302 ttak_mul_64(a.hi, b.hi, &hi_hi, &hi_lo);
303
304 uint64_t limb0 = lo_lo;
305
306 uint64_t limb1 = lo_hi;
307 uint64_t carry = 0;
308 uint64_t sum = limb1 + cross1_lo;
309 carry = (sum < limb1) ? 1 : 0;
310 limb1 = sum;
311 sum = limb1 + cross2_lo;
312 carry += (sum < limb1) ? 1 : 0;
313 limb1 = sum;
314
315 uint64_t limb2 = cross1_hi;
316 uint64_t carry2 = 0;
317 sum = limb2 + cross2_hi;
318 carry2 = (sum < limb2) ? 1 : 0;
319 limb2 = sum;
320 sum = limb2 + hi_lo;
321 carry2 += (sum < limb2) ? 1 : 0;
322 limb2 = sum;
323 limb2 += carry;
324 if (limb2 < carry) carry2++;
325
326 uint64_t limb3 = hi_hi + carry2;
327
328 return ttak_u256_from_limbs(limb3, limb2, limb1, limb0);
329}
