96 uint32_t sum = h->magic;

98 sum ^= (uint32_t)(h->created_tick >> 32);

102#if defined(__LP64__) || defined(_WIN64)

107 sum ^= (uint32_t)h->canary_start;
108 sum ^= (uint32_t)(h->canary_start >> 32);
109 sum ^= (uint32_t)h->canary_end;
